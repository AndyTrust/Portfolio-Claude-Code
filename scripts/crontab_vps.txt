# ============================================================
# Portfolio Intelligence — Crontab VPS
# Installa con: crontab crontab_vps.txt
# Timezone: Europe/Rome
#
# SCHEDULE:
#   Lun-Ven: prezzi ogni ora + geo/news ogni ora + insider giornaliero
#   Sab-Dom: solo geo/news/sentiment (mercati chiusi)
# ============================================================
SHELL=/bin/bash
PATH=/usr/local/sbin:/usr/local/bin:/sbin:/bin:/usr/sbin:/usr/bin
MAILTO=""

HOME_DIR=/root
REPO_DIR=/root/Portfolio-Claude-Code
VENV_PYTHON=/root/portfolio_venv/bin/python
LOG_DIR=/root/portfolio_logs

# ════════════════════════════════════════════════════════════
# LUNI — VENERDÌ (mercati aperti)
# ════════════════════════════════════════════════════════════

# ── Prezzi live + P&L (ogni ora :00, escluso 02-05) ─────────
0 0-1,6-23 * * 1-5 $VENV_PYTHON $REPO_DIR/scripts/vps_hourly_monitor.py >> $LOG_DIR/hourly.log 2>&1

# ── Geo/social/news (ogni ora :05, escluso 02-05) ───────────
5 0-1,6-23 * * 1-5 $VENV_PYTHON $REPO_DIR/scripts/vps_social_geo_monitor.py >> $LOG_DIR/social.log 2>&1

# ── Insider Form 4 — GIORNALIERO (ore 07:00) ────────────────
0 7 * * 1-5 $VENV_PYTHON $REPO_DIR/scripts/fetch_all_daily.py >> $LOG_DIR/insider.log 2>&1

# ── Insider leggero (fetch_insider.py) — anche ore 13:00 ────
# Doppio check infragiornaliero per catturare segnali urgenti
0 13 * * 1-5 $VENV_PYTHON $REPO_DIR/scripts/fetch_insider.py --real-only >> $LOG_DIR/insider.log 2>&1

# ── Git deploy (ogni ora :30, escluso 02-05) ────────────────
30 0-1,6-23 * * 1-5 bash $REPO_DIR/scripts/vps_git_deploy.sh >> $LOG_DIR/deploy.log 2>&1

# ── Briefing sentiment (ogni 6 ore: 06, 12, 18, 00) ────────
0 0,6,12,18 * * 1-5 $VENV_PYTHON $REPO_DIR/scripts/sentiment_news_updater.py >> $LOG_DIR/sentiment.log 2>&1

# ════════════════════════════════════════════════════════════
# SABATO — DOMENICA (mercati chiusi: no prezzi, solo news)
# ════════════════════════════════════════════════════════════

# ── Geo/social/news weekend (ogni 2 ore: 08, 10, 12, 14, 16, 18, 20) ─
0 8,10,12,14,16,18,20 * * 6,0 $VENV_PYTHON $REPO_DIR/scripts/vps_social_geo_monitor.py >> $LOG_DIR/social.log 2>&1

# ── Briefing weekend (mattina 09:00, sera 18:00) ────────────
0 9,18 * * 6,0 $VENV_PYTHON $REPO_DIR/scripts/sentiment_news_updater.py >> $LOG_DIR/sentiment.log 2>&1

# ── Git deploy weekend (ogni 2 ore :30, se ci sono news nuove) ─
30 8,10,12,14,16,18,20 * * 6,0 bash $REPO_DIR/scripts/vps_git_deploy.sh >> $LOG_DIR/deploy.log 2>&1

# ════════════════════════════════════════════════════════════
# REPORT AI — Riassunti giornaliero / settimanale / mensile
# ════════════════════════════════════════════════════════════

# ── Briefing giornaliero (Lun-Ven ore 18:00) ─────────────────
0 18 * * 1-5 $VENV_PYTHON $REPO_DIR/scripts/generate_reports.py daily >> $LOG_DIR/reports.log 2>&1

# ── Report settimanale (domenica ore 20:00) ──────────────────
0 20 * * 0 $VENV_PYTHON $REPO_DIR/scripts/generate_reports.py weekly >> $LOG_DIR/reports.log 2>&1

# ── Report mensile (1° del mese ore 20:00) ──────────────────
0 20 1 * * $VENV_PYTHON $REPO_DIR/scripts/generate_reports.py monthly >> $LOG_DIR/reports.log 2>&1

# ════════════════════════════════════════════════════════════
# MANUTENZIONE
# ════════════════════════════════════════════════════════════

# ── Cleanup log >7 giorni (domenica 03:00) ──────────────────
0 3 * * 0 find $LOG_DIR -name "*.log" -mtime +7 -delete
